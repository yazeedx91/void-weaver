## 🧪 Test Scripts

### Pre-request Script (for all requests):
```javascript
// Pre-request script for ShaheenPulse AI Stress Test
// Set up test environment variables

if (!pm.environment.get("test_start_time")) {
    pm.environment.set("test_start_time", new Date().toISOString());
}

// Generate unique test ID for this run
const testId = 'SHAHEEN_TEST_' + Date.now();
pm.environment.set("test_id", testId);

// Log test initiation
console.log('🚀 ShaheenPulse AI Stress Test Started: ' + testId);

// Set authentication token if not present
if (!pm.environment.get("auth_token")) {
    pm.environment.set("auth_token", "industrial-test-token");
}
```

### Post-response Script (for all requests):
```javascript
// Post-response script for ShaheenPulse AI Stress Test

// Log response details
console.log('📊 Response Status: ' + pm.response.code);
console.log('📊 Response Time: ' + pm.response.responseTime + 'ms');

// Parse response body if JSON
try {
    const response = pm.response.json();
    
    // Check for industrial compliance
    if (response.industrial_compliance) {
        pm.test('✅ Industrial Compliance Verified', () => {
            pm.expect(response.industrial_compliance).to.be.true;
        });
    }
    
    // Check for Phalanx Twin-Gating
    if (response.phalanx_twin_gating) {
        pm.test('🛡️ Phalanx Twin-Gating Active', () => {
            pm.expect(response.phalanx_twin_gating).to.be.true;
        });
    }
    
    // Check for Aeon Self-Healing
    if (response.aeon_self_healing) {
        pm.test('🔄 Aeon Self-Healing Triggered', () => {
            pm.expect(response.aeon_self_healing).to.be.true;
        });
    }
    
    // Check Vitality Index recovery
    if (response.vitality_index) {
        pm.test('📊 Vitality Index Within Acceptable Range', () => {
            pm.expect(response.vitality_index).to.be.above(0.7);
        });
    }
    
    // Check response time for industrial requirements
    pm.test('⚡ Response Time < 100ms', () => {
        pm.expect(pm.response.responseTime).to.be.below(100);
    });
    
} catch (e) {
    console.log('⚠️ Error parsing response:', e.message);
}

// Store test results for analysis
const testResult = {
    test_id: pm.environment.get("test_id"),
    timestamp: new Date().toISOString(),
    status_code: pm.response.code,
    response_time: pm.response.responseTime,
    success: pm.response.code >= 200 && pm.response.code < 300
};

// Store in environment for later analysis
const results = pm.environment.get("test_results") || [];
results.push(testResult);
pm.environment.set("test_results", JSON.stringify(results));

console.log('📈 Test Result:', JSON.stringify(testResult, null, 2));
```
